安装依赖
yum install epel-release -y
yum install gcc gettext autoconf libtool automake make pcre-devel asciidoc xmlto c-ares-devel libev-devel libsodium-devel mbedtls-devel wget vim  -y

安装yum源
cd  /etc/yum.repos.d/
wget https://copr.fedorainfracloud.org/coprs/librehat/shadowsocks/repo/epel-7/librehat-shadowsocks-epel-7.repo

yum -y install shadowsocks-libev

vim /etc/shadowsocks-libev/config.json

{
    修改配置
}

systemctl enable shadowsocks-libev
systemctl start shadowsocks-libev
systemctl status shadowsocks-libev
systemctl stop shadowsocks-libev


# 放通端口，注意修改
firewall-cmd --permanent --zone=public --add-port=20000/tcp
firewall-cmd --permanent --zone=public --add-port=20000/udp

# 使配置生效
firewall-cmd --reload
